(function () {
  const iframe = document.getElementById("sections-portfolio-midia-mobile-disciplinas");
  const expectedPath = "/sections/portfolio-midia-mobile/disciplinas";

  window.addEventListener("message", function (event) {
    const data = event.data;

    if (typeof data.height === "number" && data.height > 0) {
      iframe.style.height = data.height + "px";
    }
  });
})();
</script>
