const COLORS = {
    sectionBg: "#efefef",
    cardBg: "#f9f9f9",
    stroke: "#ffffff",
    text: "#121212",
    muted: "#5f5f5f",
    primary: "#3715ff",
}

const SECTION_RADIUS = 10
const HEADER_GAP = 45
const GRID_GAP = 16
const FIGMA_FRAME_HEIGHT = 410
const BASE_ICON_TARGET_HEIGHT = 74
const SECTION_PADDING_LG = 80
const SECTION_PADDING_MD = 56
const SECTION_PADDING_SM = 40
const SECTION_PADDING_XS = 32
const SECTION_PADDING_MOBILE = 20

function getIconSize(item: ObjectiveItem, targetHeight: number) {
    const height = Math.round((item.fh / FIGMA_FRAME_HEIGHT) * targetHeight)
    const width = Math.round((item.fw / FIGMA_FRAME_HEIGHT) * targetHeight)
    return { width, height }
}

function clamp(value: number, min: number, max: number) {
    return Math.min(Math.max(value, min), max)
}

function useElementSize<T extends HTMLElement>(ref: React.RefObject<T | null>) {
    const [size, setSize] = React.useState({ width: 1440, height: 980 })

    React.useLayoutEffect(() => {
        const node = ref.current
        if (!node) return

        const update = () => {
            const rect = node.getBoundingClientRect()
            setSize({
                width: rect.width || 1440,
                height: rect.height || 980,
            })
        }

        update()

        const observer = new ResizeObserver((entries) => {
            const entry = entries[0]
            if (!entry) return
            setSize({
                width: entry.contentRect.width || 1440,
                height: entry.contentRect.height || 980,
            })
        })

        observer.observe(node)
        return () => observer.disconnect()
    }, [ref])

    return size
}

export default function OQueQueremosConstruirJuntos(props: {
    style?: React.CSSProperties
}) {
    const { style } = props
    const rootRef = React.useRef<HTMLDivElement | null>(null)

    const size = useElementSize(rootRef)
    const width = size.width
    const height = size.height

    const [index, setIndex] = React.useState(0)

    const isDesktop = width >= 1200
    const isTablet = width >= 760 && width < 1200
    const visibleCards = isDesktop ? 3 : isTablet ? 2 : 1

    React.useEffect(() => {
        setIndex((current) => Math.min(current, Math.max(0, OBJECTIVES.length - visibleCards)))
    }, [visibleCards])

    const sectionPadding =
        width >= 1280
            ? SECTION_PADDING_LG
            : width >= 1024
              ? SECTION_PADDING_MD
              : width >= 768
                ? SECTION_PADDING_SM
                : width >= 640
                  ? SECTION_PADDING_XS
                  : SECTION_PADDING_MOBILE

    const titleSize = height < 620 ? 32 : height < 760 ? 38 : 44
    const subtitleSize = height < 620 ? 15 : 18
    const cardPadding = height < 620 ? 16 : 24
    const cardTitleSize = height < 620 ? 26 : height < 760 ? 32 : 40
    const cardBodySize = height < 620 ? 14 : height < 760 ? 15 : 18
    const headerGap = height < 620 ? 28 : HEADER_GAP

    const contentWidth = Math.max(320, width - sectionPadding * 2)
    const contentHeight = Math.max(360, height - sectionPadding * 2)
    const titleBlockHeight = titleSize + subtitleSize * 2 + headerGap + 20
    const availableHeight = Math.max(260, contentHeight - titleBlockHeight)

    const maxCardWidth =
        (contentWidth - GRID_GAP * (visibleCards - 1)) / visibleCards
    const cardSize = Math.max(220, Math.min(maxCardWidth, availableHeight))
    const iconTargetHeight = clamp(cardSize * 0.19, 48, BASE_ICON_TARGET_HEIGHT)

    const maxIndex = Math.max(0, OBJECTIVES.length - visibleCards)
    const step = cardSize + GRID_GAP
    const trackWidth = OBJECTIVES.length * cardSize + (OBJECTIVES.length - 1) * GRID_GAP
    const translateX = index * step

    return (
        <div
            ref={rootRef}
            style={{
                ...rootStyle,
                ...style,
            }}
        >
            <section
                style={{
                    ...sectionStyle,
                    padding: sectionPadding,
                }}
            >
                <div
                    style={{
                        ...headerStyle,
                        marginBottom: headerGap,
                    }}
                >
                    <h2
                        style={{
                            ...titleStyle,
                            fontSize: titleSize,
                        }}
                    >
                        O que queremos construir juntos
                    </h2>
                    <p
                        style={{
                            ...subtitleStyle,
                            fontSize: subtitleSize,
                        }}
                    >
                        Cinco frentes estratégicas para a Sigo crescer com marca forte.
                    </p>
                </div>

                <div
                    style={{
                        position: "relative",
                        minHeight: cardSize,
                    }}
                >
                    <button
                        type="button"
                        onClick={() => setIndex((current) => Math.max(0, current - 1))}
                        disabled={index === 0}
                        style={{
                            ...arrowStyle,
                            left: 0,
                            top: 16,
                            opacity: index === 0 ? 0.35 : 1,
                            cursor: index === 0 ? "default" : "pointer",
                        }}
                    >
                        ←
                    </button>

                    <button
                        type="button"
                        onClick={() => setIndex((current) => Math.min(maxIndex, current + 1))}
                        disabled={index === maxIndex}
                        style={{
                            ...arrowStyle,
                            right: 0,
                            top: 16,
                            opacity: index === maxIndex ? 0.35 : 1,
                            cursor: index === maxIndex ? "default" : "pointer",
                        }}
                    >
                        →
                    </button>

                    <div
                        style={{
                            overflow: "hidden",
                            width: "100%",
                        }}
                    >
                        <div
                            style={{
                                display: "flex",
                                gap: GRID_GAP,
                                width: trackWidth,
                                transform: `translateX(-${translateX}px)`,
                                transition: "transform 500ms ease",
                            }}
                        >
                            {OBJECTIVES.map((item) => {
                                const iconSize = getIconSize(item, iconTargetHeight)

                                return (
                                    <div
                                        key={item.title}
                                        style={{
                                            ...cardStyle,
                                            width: cardSize,
                                            height: cardSize,
                                            padding: cardPadding,
                                        }}
                                    >
                                        <img
                                            src={item.src}
                                            alt={item.title}
                                            width={iconSize.width}
                                            height={iconSize.height}
                                            style={{
                                                width: iconSize.width,
                                                height: iconSize.height,
                                                display: "block",
                                                objectFit: "contain",
                                                filter: "brightness(0) saturate(100%) invert(22%) sepia(88%) saturate(6150%) hue-rotate(248deg) brightness(96%)",
                                            }}
                                        />

                                        <div
                                            style={{
                                                display: "flex",
                                                flexDirection: "column",
                                                gap: 12,
                                            }}
                                        >
                                            <div
                                                style={{
                                                    ...cardTitleStyle,
                                                    fontSize: cardTitleSize,
                                                }}
                                            >
                                                {item.title}
                                            </div>
                                            <div
                                                style={{
                                                    ...cardBodyStyle,
                                                    fontSize: cardBodySize,
                                                }}
                                            >
                                                {item.body}
                                            </div>
                                        </div>
                                    </div>
                                )
                            })}
                        </div>
                    </div>
                </div>
            </section>
        </div>
    )
}
